# –¶–µ–ª—å —Ä–∞–±–æ—Ç—ã

1.  –†–∞–∑–≤–∏—Ç—å –ø—Ä–∞–∫—Ç–∏—á–µ—Å–∫–∏–µ –Ω–∞–≤—ã–∫–∏ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è —è–∑—ã–∫–∞ –ø—Ä–æ–≥—Ä–∞–º–º–∏—Ä–æ–≤–∞–Ω–∏—è R –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ –¥–∞–Ω–Ω—ã—Ö

3.  –†–∞–∑–≤–∏—Ç—å –ø—Ä–∫–∞—Ç–∏—á–µ—Å–∫–∏–µ –Ω–∞–≤—ã–∫–∏ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è —Ñ—É–Ω–∫—Ü–∏–π –æ–±—Ä–∞–±–æ—Ç–∫–∏ –¥–∞–Ω–Ω—ã—Ö –ø–∞–∫–µ—Ç–∞ dplyr -- —Ñ—É–Ω–∫—Ü–∏–∏ select(), filter(), mutate(), arrange(), group_by()

# –ó–∞–¥–∞–Ω–∏–µ

–ü—Ä–æ–∞–Ω–∞–ª–∏–∑–∏—Ä–æ–≤–∞—Ç—å –≤—Å—Ç—Ä–æ–µ–Ω–Ω—ã–π –≤ –ø–∞–∫–µ—Ç dplyr –Ω–∞–±–æ—Ä –¥–∞–Ω–Ω—ã—Ö starwars —Å –ø–æ–º–æ—â—å—é —è–∑—ã–∫–∞ R –∏ –æ—Ç–≤–µ—Ç–∏—Ç—å –Ω–∞ –≤–æ–ø—Ä–æ—Å—ã:

```{r}
library(dplyr)
```

1.–°–∫–æ–ª—å–∫–æ —Å—Ç—Ä–æ–∫ –≤ –¥–∞—Ç–∞—Ñ—Ä–µ–π–º–µ?

```{r}
starwars %>% nrow()
```

2.  –°–∫–æ–ª—å–∫–æ —Å—Ç–æ–ª–±—Ü–æ–≤ –≤ –¥–∞—Ç–∞—Ñ—Ä–µ–π–º–µ?

```{r}
starwars %>%  ncol()
```

3.  –ö–∞–∫ –ø—Ä–æ—Å–º–æ—Ç—Ä–µ—Ç—å –ø—Ä–∏–º–µ—Ä–Ω—ã–π –≤–∏–¥ –¥–∞—Ç–∞—Ñ—Ä–µ–π–º–∞?

```{r}
starwars %>% glimpse()
```

4.  –°–∫–æ–ª—å–∫–æ —É–Ω–∏–∫–∞–ª—å–Ω—ã—Ö —Ä–∞—Å –ø–µ—Ä—Å–æ–Ω–∞–∂–µ–π (species) –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–æ –≤ –¥–∞–Ω–Ω—ã—Ö?

```{r}
length(unique(starwars$species))
```

5.  –ù–∞–π—Ç–∏ —Å–∞–º–æ–≥–æ –≤—ã—Å–æ–∫–æ–≥–æ –ø–µ—Ä—Å–æ–Ω–∞–∂–∞.

```{r}
arrange(starwars, desc(height), by_group=TRUE) %>% slice(1:1) %>% select(name:height)
```

6.  –ù–∞–π—Ç–∏ –≤—Å–µ—Ö –ø–µ—Ä—Å–æ–Ω–∞–∂–µ–π –Ω–∏–∂–µ 170

```{r}
filter(starwars, height < 170) %>% pull(name)
```

7.  –ü–æ–¥—Å—á–∏—Ç–∞—Ç—å –ò–ú–¢ (–∏–Ω–¥–µ–∫—Å –º–∞—Å—Å—ã —Ç–µ–ª–∞) –¥–ª—è –≤—Å–µ—Ö –ø–µ—Ä—Å–æ–Ω–∞–∂–µ–π. –ò–ú–¢ –ø–æ–¥—Å—á–∏—Ç–∞—Ç—å –ø–æ —Ñ–æ—Ä–º—É–ª–µ ùêº = ùëö/‚Ñé\^2, –≥–¥–µ ùëö -- –º–∞—Å—Å–∞ (weight), –∞ ‚Ñé -- —Ä–æ—Å—Ç (height).

```{r}
starwars %>% mutate(IMT = mass / height^2) %>% select(name,IMT)
```

8.  –ù–∞–π—Ç–∏ 10 —Å–∞–º—ã—Ö "–≤—ã—Ç—è–Ω—É—Ç—ã—Ö" –ø–µ—Ä—Å–æ–Ω–∞–∂–µ–π. "–í—ã—Ç—è–Ω—É—Ç–æ—Å—Ç—å" –æ—Ü–µ–Ω–∏—Ç—å –ø–æ –æ—Ç–Ω–æ—à–µ–Ω–∏—é –º–∞—Å—Å—ã (mass) –∫ —Ä–æ—Å—Ç—É(height) –ø–µ—Ä—Å–æ–Ω–∞–∂–µ–π.

```{r}
starwars %>% mutate(–í—ã—Ç—è–Ω—É—Ç–æ—Å—Ç—å = mass/height) %>% arrange(desc(–í—ã—Ç—è–Ω—É—Ç–æ—Å—Ç—å), by_group=TRUE) %>% select(name,–í—ã—Ç—è–Ω—É—Ç–æ—Å—Ç—å)
```

9.  –ù–∞–π—Ç–∏ —Å—Ä–µ–¥–Ω–∏–π –≤–æ–∑—Ä–∞—Å—Ç –ø–µ—Ä—Å–æ–Ω–∞–∂–µ–π –∫–∞–∂–¥–æ–π —Ä–∞—Å—ã –≤—Å–µ–ª–µ–Ω–Ω–æ–π –ó–≤–µ–∑–¥–Ω—ã—Ö –≤–æ–π–Ω.

```{r}
starwars %>% group_by(species) %>% filter(!is.na(birth_year)) %>% summarise(MidAge = mean(birth_year, na.rm = TRUE)) %>% select(species,MidAge) %>% arrange(desc(MidAge))
```

10. –ù–∞–π—Ç–∏ —Å–∞–º—ã–π —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–Ω—ã–π —Ü–≤–µ—Ç –≥–ª–∞–∑ –ø–µ—Ä—Å–æ–Ω–∞–∂–µ–π –≤—Å–µ–ª–µ–Ω–Ω–æ–π –ó–≤–µ–∑–¥–Ω—ã—Ö –≤–æ–π–Ω.

```{r}
starwars %>% group_by(eye_color) %>% summarise(–ö–æ–ª–∏—á–µ—Å—Ç–≤–æ = n()) %>% arrange(desc(–ö–æ–ª–∏—á–µ—Å—Ç–≤–æ)) %>% slice(1:1)
```

11. –ü–æ–¥—Å—á–∏—Ç–∞—Ç—å —Å—Ä–µ–¥–Ω—é—é –¥–ª–∏–Ω—É –∏–º–µ–Ω–∏ –≤ –∫–∞–∂–¥–æ–π —Ä–∞—Å–µ –≤—Å–µ–ª–µ–Ω–Ω–æ–π –ó–≤–µ–∑–¥–Ω—ã—Ö –≤–æ–π–Ω.

```{r}
starwars %>% group_by(species) %>% mutate(NL = nchar(name)) %>% summarise(nameleng = mean(NL)) %>% select(species, nameleng) %>% arrange(desc(nameleng))
```
